---
title: 'Create bot'
description: 'Create a new OpenClaw bot instance for a user. Returns the bot record including its auto-generated access token and endpoint slug.'
---

`POST /bot/api/v1/bots`

<ParamField header="Authorization" type="string" required>
  Bearer token for the app. Format: `Bearer <app-token>`
</ParamField>

## Request body

<ParamField body="user_id" type="string" required>
  ID of the user who owns this bot.
</ParamField>

<ParamField body="name" type="string" required>
  Display name for the bot.
</ParamField>

<ParamField body="slug" type="string">
  URL-friendly identifier used for subdomain routing (e.g., `my-bot` becomes `my-bot.clawhost.ai`). Must be 1–50 characters: lowercase letters, numbers, and hyphens only. Auto-generated from the bot ID if omitted.
</ParamField>

<ParamField body="config" type="object">
  OpenClaw configuration for the bot.

  <Expandable title="config properties">
    <ParamField body="provider" type="string">
      Legacy single-provider key (e.g., `anthropic`, `openai`, `minimax`). Use `providers[]` for multi-provider setups.
    </ParamField>

    <ParamField body="model" type="string">
      Legacy default model ID (e.g., `claude-sonnet-4-20250514`). Use `providers[].models[]` for multi-provider setups.
    </ParamField>

    <ParamField body="api_key" type="string">
      API key for the legacy single provider.
    </ParamField>

    <ParamField body="base_url" type="string">
      Custom base URL for Anthropic-compatible APIs (e.g., MiniMax).
    </ParamField>

    <ParamField body="auth" type="string">
      Auth mode for the legacy provider. Defaults to `api-key`. Options: `api-key`, `bearer`.
    </ParamField>

    <ParamField body="api" type="string">
      API format for the legacy provider. Defaults to `anthropic-messages`. Options: `anthropic-messages`, `openai-completions`.
    </ParamField>

    <ParamField body="providers" type="object[]">
      List of model provider configurations for multi-provider setups.

      <Expandable title="providers[] properties">
        <ParamField body="name" type="string" required>
          Provider name (e.g., `anthropic`, `openai`, `minimax`).
        </ParamField>
        <ParamField body="api_key" type="string">
          API key for this provider.
        </ParamField>
        <ParamField body="base_url" type="string">
          Custom base URL for this provider.
        </ParamField>
        <ParamField body="auth" type="string">
          Auth mode: `api-key` or `bearer`.
        </ParamField>
        <ParamField body="api" type="string">
          API format: `anthropic-messages` or `openai-completions`.
        </ParamField>
        <ParamField body="models" type="object[]">
          Model configurations for this provider.

          <Expandable title="models[] properties">
            <ParamField body="id" type="string" required>
              Model ID (e.g., `claude-sonnet-4-20250514`).
            </ParamField>
            <ParamField body="name" type="string">
              Human-readable model name.
            </ParamField>
            <ParamField body="reasoning" type="boolean">
              Whether the model supports extended reasoning.
            </ParamField>
            <ParamField body="input" type="string[]">
              Supported input modalities (e.g., `["text"]`, `["text", "image"]`).
            </ParamField>
            <ParamField body="context_window" type="number">
              Context window size in tokens.
            </ParamField>
            <ParamField body="max_tokens" type="number">
              Maximum output tokens.
            </ParamField>
          </Expandable>
        </ParamField>
      </Expandable>
    </ParamField>

    <ParamField body="agent_defaults" type="object">
      Default model assignments for agents.

      <Expandable title="agent_defaults properties">
        <ParamField body="primary_model" type="string">
          Primary model ID in `provider/model-id` format (e.g., `anthropic/claude-sonnet-4-20250514`).
        </ParamField>
        <ParamField body="fallback_model" type="string">
          Fallback model ID used when the primary model is unavailable.
        </ParamField>
      </Expandable>
    </ParamField>

    <ParamField body="mcp_servers" type="object[]">
      MCP (Model Context Protocol) server definitions.

      <Expandable title="mcp_servers[] properties">
        <ParamField body="name" type="string" required>
          Server identifier.
        </ParamField>
        <ParamField body="command" type="string" required>
          Executable command to launch the MCP server.
        </ParamField>
        <ParamField body="args" type="string[]">
          Command-line arguments.
        </ParamField>
        <ParamField body="env" type="object">
          Environment variables as key-value pairs.
        </ParamField>
        <ParamField body="share_process" type="boolean">
          Share the process across agent sessions.
        </ParamField>
      </Expandable>
    </ParamField>

    <ParamField body="agents_md" type="string">
      Markdown content that defines available agents.
    </ParamField>

    <ParamField body="soul_md" type="string">
      Markdown content that defines the bot's personality and system prompt.
    </ParamField>

    <ParamField body="tools_md" type="string">
      Markdown content that defines available tools.
    </ParamField>

    <ParamField body="channels" type="object">
      IM channel configuration. Keys are channel names (`telegram`, `slack`, `discord`, etc.), values are channel-specific account configs.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="expires_at" type="string">
  ISO 8601 timestamp after which the bot expires. Omit for no expiry.
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique bot UUID.
</ResponseField>

<ResponseField name="app_id" type="string">
  ID of the app that owns this bot.
</ResponseField>

<ResponseField name="user_id" type="string">
  ID of the user who owns this bot.
</ResponseField>

<ResponseField name="name" type="string">
  Display name.
</ResponseField>

<ResponseField name="slug" type="string">
  URL-friendly identifier used in subdomain routing.
</ResponseField>

<ResponseField name="access_token" type="string">
  Cryptographically secure token used for CLI commands and bot-level authentication.
</ResponseField>

<ResponseField name="status" type="string">
  Bot status. Newly created bots always start with `created`.
</ResponseField>

<ResponseField name="config" type="object">
  Full OpenClaw configuration stored for this bot.
</ResponseField>

<ResponseField name="endpoint" type="string">
  Internal K8s service endpoint. Empty until the bot is started.
</ResponseField>

<ResponseField name="access_url" type="string">
  External access URL in the form `https://{slug}.{bot-domain}?token={access_token}`.
</ResponseField>

<ResponseField name="expires_at" type="string">
  Expiry timestamp, or `null` if the bot never expires.
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 creation timestamp.
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 last-updated timestamp.
</ResponseField>

<RequestExample>

```bash cURL
curl -s -X POST http://<host>:18080/bot/api/v1/bots \
  -H "Authorization: Bearer $API_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "user_id": "user-001",
    "name": "my-bot",
    "slug": "my-bot",
    "config": {
      "model": "claude-sonnet-4-20250514",
      "api_key": "sk-ant-xxx"
    }
  }'
```

</RequestExample>
